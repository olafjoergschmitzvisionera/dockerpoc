Run gitlab

docker pull gitlab/gitlab-ce:latest

then export GITLAB_HOME=~/docker

mkdir config

mkdir log

mkdir data

sudo docker run --detach --hostname gitlab.example.com --restart always --publish 443:443 --publish 80:80 --publish 22:22   --name gitlab   --restart always \ --volume $GITLAB_HOME/config:/etc/gitlab:Z --volume $GITLAB_HOME/log:/var/log/gitlab:Z --volume $GITLAB_HOME/data:/var/log/gitlab:Z gitlab/gitlab-ce:latest
